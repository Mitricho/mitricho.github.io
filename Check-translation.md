## üõ†Ô∏è –ò—Ç–æ–≥–æ–≤—ã–π –∏–Ω—Å—Ç—Ä—É–º–µ–Ω—Ç: —Å–∫—Ä–∏–ø—Ç –Ω–∞ Python

```python
# check_translation.py
import pandas as pd
import re
from sentence_transformers import SentenceTransformer
from sklearn.metrics.pairwise import cosine_similarity

def is_chinese(text):
    return bool(re.search(r'[\u4e00-\u9fff]', str(text)))

def load_md_table(path):
    df = pd.read_csv(path, sep='|', skipinitialspace=True)
    df = df.drop(df.columns[[0, -1]], axis=1)
    df.columns = df.columns.str.strip()
    return df[['–†—É—Å—Å–∫–∏–π', '–ö–∏—Ç–∞–π—Å–∫–∏–π']]

def main(md_path, threshold=0.7):
    df = load_md_table(md_path)
    df['is_chinese'] = df['–ö–∏—Ç–∞–π—Å–∫–∏–π'].apply(is_chinese)
    df = df[df['is_chinese']].copy()

    model = SentenceTransformer('sentence-transformers/paraphrase-multilingual-MiniLM-L12-v2')
    rus_emb = model.encode(df['–†—É—Å—Å–∫–∏–π'].tolist())
    zh_emb = model.encode(df['–ö–∏—Ç–∞–π—Å–∫–∏–π'].tolist())
    sims = [cosine_similarity([r], [z])[0][0] for r, z in zip(rus_emb, zh_emb)]
    df['Similarity'] = sims
    df['NeedsReview'] = df['Similarity'] < threshold

    # –°–æ—Ö—Ä–∞–Ω–∏—Ç—å –æ—Ç—á—ë—Ç
    df.to_csv('translation_check_report.csv', index=False)
    print(f"–ü—Ä–æ–≤–µ—Ä–∫–∞ –∑–∞–≤–µ—Ä—à–µ–Ω–∞. –ü–æ–¥–æ–∑—Ä–∏—Ç–µ–ª—å–Ω—ã—Ö —Å—Ç—Ä–æ–∫: {df['NeedsReview'].sum()}")
```
